### W2.3 — Idempotency Keys + decision_basis_hash (1 Tag)

def compute_decision_basis_hash(state: SealAIState) -> str:
    relevant = {
        "normalized": state.normalized.dict(),
        "derived": state.derived.dict(),
        "evidence_versions": state.evidence.source_versions,
        "ontology_version": STS_VERSION,
        "prompt_version": PROMPT_VERSION,
    }
    return hashlib.sha256(json.dumps(relevant, sort_keys=True).encode()).hexdigest()[:16]

```python
async def generate_inquiry_pdf(state: SealAIState) -> bytes:
    # Idempotency-Check
    key = state.action_readiness.idempotency_key
    cached = await redis.get(f"pdf:{key}")
    if cached: return cached

    # Jinja2 → HTML
    html = prompts.render("pdf/inquiry.html.j2", build_pdf_context(state))

    # Gotenberg → PDF
    async with httpx.AsyncClient() as client:
        response = await client.post(
            f"{settings.GOTENBERG_URL}/forms/chromium/convert/html",
            files={"files": ("index.html", html, "text/html")},
            timeout=30.0
        )

    # Cache + Audit
    await redis.setex(f"pdf:{key}", 86400, response.content)
    await write_pdf_audit(state, key)
    return response.content
```

### W4.4 — interrupt() für Clarification-Loop (2–3 Tage)

```python
# graph/nodes/renderer_node.py
from langgraph.types import interrupt

async def renderer_node(state: SealAIState) -> SealAIState:
    context = build_renderer_context(state)
    system_prompt = prompts.renderer_system_prompt(
        outward_class=state.decision.outward_class,
        context=context
    )
    response = await llm.ainvoke(system_prompt)

    if state.decision.outward_class == "structured_clarification":
        user_input = interrupt({"response": response.content})
        return state.model_copy(update={
            "observed": state.observed.model_copy(
                update={"raw_input": user_input}
            )
        })

    return state.model_copy(update={
        "action_readiness": state.action_readiness.model_copy(update={
            "rendered_response": {
                "outward_class": state.decision.outward_class,
                "content": response.content,
            }
        })
    })
```
